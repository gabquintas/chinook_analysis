import sqlite3
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

% matplotlib inline

def import_query (script):
    """
    This function goal is to import a query result from
    a database and convert it into a pandas dataframe.
    Imput: a script written in SQL language
    Output: the script result in dataframe format
    """
    conn = sqlite3.connect("/Users/Gavrik/Desktop/Programming/Udacity/Data_Science_Fundamentals_II/chinook_db/chinook.db")
    cursor = conn.cursor()
    cursor.execute(script)
    df_columns = [column[0] for column in cursor.description]
    results = cursor.fetchall()
    conn.close()    
    return pd.DataFrame(results, columns = df_columns)
